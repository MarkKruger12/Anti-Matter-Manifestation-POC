import numpy as np
import pandas as pd
from datetime import datetime
from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
from scipy.optimize import dual_annealing
import random
import time

# --- CONFIGURATION ---
NUM_RANGE = 49  # Max possible number in the draw (1 to 49)
NUM_DRAWS = 1000 # Number of historical draws to simulate for the Tesseract's training
N_COMPONENTS = 3 # Dimensionality of the structural map
NUM_SA_ITERATIONS = 5  # Number of separate Simulated Annealing runs to perform
SEED = 42
# ---------------------

np.random.seed(SEED)
random.seed(SEED)

# Global variables for Tesseract Core access
global_scaler = None
global_pca_model = None
global_pca_center = None

# ---------- Helpers ----------
def continuous_to_unique_numbers(x, low=1, high=NUM_RANGE, k=6):
    """
    Convert a continuous 6-vector x into k unique integers in [low, high].
    Deterministic tie-breaker: uses original fractional parts to nudge duplicates.
    Returns sorted array of length k of ints.
    """
    # ensure numpy
    x = np.asarray(x, dtype=float).flatten()
    # clip to bounds
    x = np.clip(x, low, high)
    # initial rounding
    rounded = np.round(x).astype(int)
    # We will resolve duplicates deterministically by nudging based on fractional parts
    frac = x - np.floor(x)
    # if duplicates exist, adjust iteratively
    seen = {}
    for i in range(len(rounded)):
        val = int(rounded[i])
        # try to find unique slot by nudging up/down deterministically
        if val not in seen:
            seen[val] = [i]
            continue
        # if duplicate, attempt to find nearest unused integer based on frac sign
        direction = -1 if frac[i] < 0.5 else 1
        candidate = val
        attempt = 0
        while True:
            attempt += 1
            candidate = candidate + direction * attempt
            # wrap inside [low, high]
            if candidate < low:
                candidate = low
            if candidate > high:
                candidate = high
            if candidate not in seen:
                rounded[i] = candidate
                seen.setdefault(candidate, []).append(i)
                break
            # if couldn't find in that direction, flip direction
            if attempt > (high - low):
                # fallback: linear scan for first available
                for cand in range(low, high+1):
                    if cand not in seen:
                        rounded[i] = cand
                        seen.setdefault(cand, []).append(i)
                        break
                break

    # Final safety: if we somehow have duplicates still, use unique selection by order
    unique_vals = []
    for v in rounded:
        if v not in unique_vals:
            unique_vals.append(v)
    # if too few (shouldn't happen), fill with lowest unused ints
    if len(unique_vals) < k:
        for cand in range(low, high + 1):
            if cand not in unique_vals:
                unique_vals.append(cand)
            if len(unique_vals) == k:
                break

    return np.sort(np.array(unique_vals[:k], dtype=int))


def numbers_to_ohe(numbers, num_range=NUM_RANGE):
    """Converts a list of numbers to the 49D OHE vector. Accepts iterables of ints."""
    try:
        unique_numbers = np.unique(np.array(numbers, dtype=int))
    except Exception:
        return None

    if len(unique_numbers) != 6:
        return None

    if np.any(unique_numbers < 1) or np.any(unique_numbers > num_range):
        return None

    vector = np.zeros(num_range, dtype=int)
    for num in unique_numbers:
        vector[num - 1] = 1

    if np.sum(vector) != 6:
        return None

    return vector

# ====================================================================
# STEP 1: SENSORY LAYER (unchanged)
# ====================================================================
def generate_synthetic_data(num_draws, num_range=NUM_RANGE, numbers_per_draw=6):
    data = []
    dates = pd.date_range(end=datetime.now(), periods=num_draws, freq='3D')

    for date in dates:
        draw = np.sort(np.random.choice(range(1, num_range + 1), numbers_per_draw, replace=False))
        vector = np.zeros(num_range, dtype=int)
        for num in draw:
            vector[num - 1] = 1
        data.append({
            'Draw Date': date,
            'Draw_Numbers_Sorted': draw.tolist(),
            'Vector_OHE': vector
        })
    return pd.DataFrame(data)

# ====================================================================
# STEP 2: PATTERN LAYER (PCA) (unchanged)
# ====================================================================
def run_pattern_layer(df_input, X_vectors):
    global global_scaler, global_pca_model, global_pca_center
    global_scaler = StandardScaler()
    X_scaled = global_scaler.fit_transform(X_vectors)
    global_pca_model = PCA(n_components=N_COMPONENTS, random_state=SEED)
    global_pca_model.fit(X_scaled)
    X_pca = global_pca_model.transform(X_scaled)
    global_pca_center = X_pca.mean(axis=0)
    total_explained = global_pca_model.explained_variance_ratio_.sum() * 100
    print(f"✅ PCA Complete. Total Variance Explained: {total_explained:.2f}%")
    print("Structural Center:", global_pca_center)
    return df_input

# ====================================================================
# STEP 3: POSSIBILITY LAYER (Simulated Annealing V2) - improved
# ====================================================================
def energy_function_v2_continuous(x):
    """
    Accepts continuous 6-vector x from dual_annealing.
    Converts deterministically to 6 unique integers and returns squared distance
    from the PCA center as the energy.
    """
    # map to 6 unique ints deterministically
    numbers = continuous_to_unique_numbers(x, low=1, high=NUM_RANGE, k=6)
    ohe_vector = numbers_to_ohe(numbers)
    if ohe_vector is None:
        return 1e9  # very high penalty

    try:
        scaled_vector = global_scaler.transform(ohe_vector.reshape(1, -1))
        pca_vector = global_pca_model.transform(scaled_vector)  # shape (1, N_COMPONENTS)
    except Exception:
        return 1e9

    # squared Euclidean distance to coherence anchor
    distance_squared = float(np.sum((pca_vector.flatten() - global_pca_center.flatten())**2))
    return distance_squared


def run_possibility_layer(num_runs=NUM_SA_ITERATIONS):
    print("\n--- STEP 3: POSSIBILITY LAYER (Simulated Annealing V2) ---")
    bounds = [(1, NUM_RANGE)] * 6
    coherent_draws = []

    for i in range(num_runs):
        start_time = time.time()
        print(f"  > Running Annealing Search {i+1}/{num_runs}...")
        result = dual_annealing(func=energy_function_v2_continuous, bounds=bounds, maxiter=1000, seed=SEED)
        end_time = time.time()

        # map result.x to unique numbers deterministically
        numbers = continuous_to_unique_numbers(result.x, low=1, high=NUM_RANGE, k=6)
        final_ohe = numbers_to_ohe(numbers)

        if final_ohe is not None:
            final_energy = energy_function_v2_continuous(result.x)
            coherent_draws.append({
                'Run': i + 1,
                'Coherence_Energy': final_energy,
                'Draw_Numbers': sorted(numbers.tolist()),
                'Time_Seconds': round(end_time - start_time, 2),
                'OHE_Vector': final_ohe
            })
        else:
            coherent_draws.append({
                'Run': i + 1,
                'Coherence_Energy': 1e6,
                'Draw_Numbers': 'N/A',
                'Time_Seconds': round(end_time - start_time, 2),
                'OHE_Vector': None
            })

    df_coherent = pd.DataFrame(coherent_draws)
    print("\n🎉 Possibility Layer Complete! Structurally Coherent Draws Found:")
    print(df_coherent.to_markdown(index=False, floatfmt=".4f"))
    return df_coherent

# ====================================================================
# STEP 4: INTEGRATION LAYER (Tesseract Core)
# ====================================================================
def run_integration_layer(df_coherent):
    print("\n--- STEP 4: INTEGRATION LAYER (Tesseract Core) ---")
    # Keep only non-None OHE vectors
    valid_vectors = [v for v in df_coherent['OHE_Vector'] if v is not None]
    if len(valid_vectors) == 0:
        print("❌ ERROR: No valid coherent draws were found for synthesis.")
        return

    avg_coherent_vector = np.mean(np.stack(valid_vectors), axis=0)
    ideal_indices = np.argsort(avg_coherent_vector)[-6:]
    structural_ideal_draw = sorted((ideal_indices + 1).tolist())

    print("\n🔮 TESSERACT CORE SYNTHESIS COMPLETE!")
    print("--------------------------------------")
    print("Structural Ideal Draw (Based on Coherence Anchor):")
    print(f"**{structural_ideal_draw}**")
    print("--------------------------------------")
    return structural_ideal_draw

if __name__ == "__main__":
    # Step 1
    df_data = generate_synthetic_data(NUM_DRAWS)
    X_vectors = np.stack(df_data['Vector_OHE'].values)
    print("Sensory data shape:", X_vectors.shape)

    # Step 2
    run_pattern_layer(df_data, X_vectors)

    # Step 3
    df_coherent_draws = run_possibility_layer()

    # Step 4
    run_integration_layer(df_coherent_draws)
