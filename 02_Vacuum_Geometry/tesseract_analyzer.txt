"""
TESSERACT LOTTO 6/49 ANALYZER
Advanced analysis and projection system
"""

import pandas as pd
import numpy as np
from datetime import datetime, timedelta
import matplotlib.pyplot as plt
from collections import Counter, defaultdict
import random

class TesseractAnalyzer:
    def __init__(self):
        self.df = pd.read_csv('lotto_649_complete.csv')
        print(f"🔮 TESSERACT ANALYSIS SYSTEM LOADED")
        print(f"📊 Dataset: {len(self.df):,} draws from {self.df.iloc[0]['Date']} to {self.df.iloc[-1]['Date']}")
        
    def basic_statistics(self):
        """Basic number frequency analysis"""
        print("\n" + "="*60)
        print("BASIC NUMBER FREQUENCY ANALYSIS")
        print("="*60)
        
        # Combine all main numbers
        all_numbers = []
        for i in range(1, 7):
            all_numbers.extend(self.df[f'Num{i}'].tolist())
        
        freq = Counter(all_numbers)
        
        print("\n🎯 MOST FREQUENT NUMBERS (All Time):")
        for num, count in freq.most_common(15):
            percentage = (count / len(self.df)) * 100
            print(f"  #{num:2d}: {count:3d} times ({percentage:.1f}%)")
        
        print("\n🥶 LEAST FREQUENT NUMBERS (All Time):")
        for num, count in freq.most_common()[-15:]:
            percentage = (count / len(self.df)) * 100
            print(f"  #{num:2d}: {count:3d} times ({percentage:.1f}%)")
        
        return freq
    
    def recent_trends(self, periods=100):
        """Analyze recent trends vs historical"""
        print(f"\n" + "="*60)
        print(f"RECENT TRENDS (Last {periods} Draws)")
        print("="*60)
        
        recent = self.df.tail(periods)
        historical = self.df.head(len(self.df) - periods)
        
        # Recent frequency
        recent_nums = []
        for i in range(1, 7):
            recent_nums.extend(recent[f'Num{i}'].tolist())
        recent_freq = Counter(recent_nums)
        
        # Historical frequency
        historical_nums = []
        for i in range(1, 7):
            historical_nums.extend(historical[f'Num{i}'].tolist())
        historical_freq = Counter(historical_nums)
        
        print("\n🔥 HOT NUMBERS (Recently Frequent):")
        hot_numbers = []
        for num, count in recent_freq.most_common(10):
            historical_count = historical_freq.get(num, 0)
            historical_avg = historical_count / len(historical) if len(historical) > 0 else 0
            recent_avg = count / len(recent)
            trend = "📈" if recent_avg > historical_avg else "➡️"
            hot_numbers.append(num)
            print(f"  #{num:2d}: {count:2d} times {trend}")
        
        print("\n❄️  COLD NUMBERS (Recently Rare):")
        cold_numbers = []
        for num, count in recent_freq.most_common()[-10:]:
            historical_count = historical_freq.get(num, 0)
            historical_avg = historical_count / len(historical) if len(historical) > 0 else 0
            recent_avg = count / len(recent)
            trend = "📉" if recent_avg < historical_avg else "➡️"
            cold_numbers.append(num)
            print(f"  #{num:2d}: {count:2d} times {trend}")
        
        return hot_numbers, cold_numbers
    
    def number_pair_analysis(self):
        """Analyze frequently occurring number pairs"""
        print(f"\n" + "="*60)
        print("NUMBER PAIR ANALYSIS")
        print("="*60)
        
        pair_count = defaultdict(int)
        
        for _, row in self.df.iterrows():
            numbers = sorted([row[f'Num{i}'] for i in range(1, 7)])
            # Count all pairs in this draw
            for i in range(len(numbers)):
                for j in range(i + 1, len(numbers)):
                    pair = (numbers[i], numbers[j])
                    pair_count[pair] += 1
        
        print("\n🤝 MOST FREQUENT NUMBER PAIRS:")
        for pair, count in sorted(pair_count.items(), key=lambda x: x[1], reverse=True)[:10]:
            percentage = (count / len(self.df)) * 100
            print(f"  {pair[0]:2d} & {pair[1]:2d}: {count:3d} times ({percentage:.1f}%)")
        
        return pair_count
    
    def generate_tesseract_projection(self, upcoming_date="November 22, 2025"):
        """Generate Tesseract projection for upcoming draw"""
        print(f"\n" + "="*60)
        print(f"🔮 TESSERACT PROJECTION FOR {upcoming_date}")
        print("="*60)
        
        # Get analysis data
        freq = self.basic_statistics()
        hot_nums, cold_nums = self.recent_trends()
        pair_count = self.number_pair_analysis()
        
        # Strategy 1: Balanced approach (hot + overdue)
        print(f"\n🎯 STRATEGY 1: BALANCED PROJECTION")
        balanced_picks = self._balanced_strategy(freq, hot_nums, cold_nums)
        print(f"   Numbers: {balanced_picks}")
        
        # Strategy 2: Hot numbers focus
        print(f"\n🔥 STRATEGY 2: HOT NUMBERS FOCUS")
        hot_picks = self._hot_number_strategy(hot_nums, freq)
        print(f"   Numbers: {hot_picks}")
        
        # Strategy 3: Overdue numbers
        print(f"\n⏰ STRATEGY 3: OVERDUE NUMBERS")
        overdue_picks = self._overdue_strategy(freq, self.df)
        print(f"   Numbers: {overdue_picks}")
        
        # Strategy 4: AI-inspired random with weights
        print(f"\n🤖 STRATEGY 4: AI-WEIGHTED SELECTION")
        ai_picks = self._ai_weighted_strategy(freq, hot_nums)
        print(f"   Numbers: {ai_picks}")
        
        return {
            'balanced': balanced_picks,
            'hot_focus': hot_picks,
            'overdue': overdue_picks,
            'ai_weighted': ai_picks
        }
    
    def _balanced_strategy(self, freq, hot_nums, cold_nums):
        """Balanced strategy mixing hot and overdue numbers"""
        # Take 3 hot numbers and 3 cold/overdue numbers
        picks = hot_nums[:3] + cold_nums[:3]
        return sorted(picks)
    
    def _hot_number_strategy(self, hot_nums, freq):
        """Focus on recently hot numbers"""
        return sorted(hot_nums[:6])
    
    def _overdue_strategy(self, freq, df):
        """Focus on numbers that are overdue"""
        # Find last occurrence of each number
        last_occurrence = {}
        for num in range(1, 50):
            mask = (df['Num1'] == num) | (df['Num2'] == num) | (df['Num3'] == num) | \
                   (df['Num4'] == num) | (df['Num5'] == num) | (df['Num6'] == num)
            occurrences = df[mask]
            if len(occurrences) > 0:
                last_occurrence[num] = len(df) - occurrences.index[-1] - 1
        
        # Get most overdue numbers
        overdue = sorted(last_occurrence.items(), key=lambda x: x[1], reverse=True)[:6]
        return sorted([num for num, _ in overdue])
    
    def _ai_weighted_strategy(self, freq, hot_nums):
        """AI-inspired weighted random selection"""
        weights = []
        for num in range(1, 50):
            # Weight based on frequency and recent heat
            base_weight = freq.get(num, 0)
            heat_boost = 10 if num in hot_nums[:10] else 0
            total_weight = base_weight + heat_boost
            weights.append(total_weight)
        
        # Normalize weights
        total = sum(weights)
        weights = [w/total for w in weights]
        
        # Weighted random selection without replacement
        numbers = list(range(1, 50))
        picks = []
        for _ in range(6):
            if not numbers:
                break
            chosen = random.choices(numbers, weights=weights, k=1)[0]
            picks.append(chosen)
            idx = numbers.index(chosen)
            numbers.pop(idx)
            weights.pop(idx)
        
        return sorted(picks)
    
    def analyze_last_draw(self):
        """Analyze the most recent actual draw"""
        print(f"\n" + "="*60)
        print("ANALYSIS OF LAST ACTUAL DRAW")
        print("="*60)
        
        last_draw = self.df.iloc[-1]
        print(f"📅 Date: {last_draw['Date']}")
        print(f"🔢 Numbers: {last_draw['Num1']}, {last_draw['Num2']}, {last_draw['Num3']}, {last_draw['Num4']}, {last_draw['Num5']}, {last_draw['Num6']}")
        print(f"🎯 Bonus: {last_draw['Bonus']}")
        
        # Analyze the pattern
        numbers = [last_draw[f'Num{i}'] for i in range(1, 7)]
        print(f"📊 Pattern Analysis:")
        print(f"   - Sum: {sum(numbers)}")
        print(f"   - Average: {sum(numbers)/6:.1f}")
        print(f"   - Range: {max(numbers) - min(numbers)}")
        print(f"   - Odd/Even: {sum(1 for n in numbers if n % 2 == 0)} even, {sum(1 for n in numbers if n % 2 == 1)} odd")
